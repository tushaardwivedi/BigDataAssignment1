class Main{		public static boolean isMatching(char[] str, char[] pattern, int n, int m, 									boolean[][] lookup)	{				if (m < 0 && n < 0) {			return true;		}			else if (m < 0) {			return false;		}		else if (n < 0)		{			for (int i = 0; i <= m; i++) {				if (pattern[i] != '*') {					return false;				}			}			return true;		}		if (!lookup[n][m])		{			if (pattern[m] == '*')			{				lookup[n][m] = isMatching(str, pattern, n - 1, m, lookup) ||							isMatching(str, pattern, n, m - 1, lookup);			}			else {				if (pattern[m] != '?' && pattern[m] != str[n]) {					lookup[n][m] = false;				}							else {					lookup[n][m] = isMatching(str, pattern, n - 1, m - 1, lookup);				}			}		}		return lookup[n][m];	}	public static void main(String[] args)	{		char[] str = "xyxzzxy".toCharArray();		char[] pattern = "x***x?".toCharArray();			boolean[][] lookup = new boolean[str.length + 1][pattern.length + 1];		if (isMatching(str, pattern, str.length - 1, pattern.length - 1, lookup)) {			System.out.print("Match");		} else {			System.out.print("No Match");		}	}}
